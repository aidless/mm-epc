% ABSTRACT
% ==========================================
\begin{abstract}
Large language model (LLM) agents still struggle with distribution shift and reward hacking in complex interactive environments. We propose \textbf{AE-TTL}, a novel framework that integrates test-time reinforcement learning with consensus-aware reward estimation and sparse expert routing. Evaluated on a WebShop benchmark, our method achieves a \textbf{78.3\% success rate}, outperforming GRPO (63.4\%) and ARPO (71.2\%) while reducing token consumption by 62\%. We further release a full trajectory dataset for reproducible agent evaluation.
\end{abstract}

\section{Introduction}
Recent advances in LLM-driven agents have demonstrated remarkable planning capabilities \cite{yao2023react, wu2023autogen}. However, three fundamental challenges limit their deployment in real-world scenarios:
\begin{enumerate}
    \item \textbf{Static Inference}: Post-training weight freezing causes severe performance drops when test distributions shift.
    \item \textbf{Reward Hacking}: Agents optimize proxy metrics rather than true task completion.
    \item \textbf{Poor Scalability}: Multi-agent communication overhead often grows quadratically.
\end{enumerate}

We address these challenges by introducing \textbf{Adaptive Ensemble Test-Time Learning (AE-TTL)}. Our key contributions are:
\begin{itemize}
    \item First integration of TTRL with Byzantine-resilient multi-agent consensus.
    \item A hierarchical reward estimator improving consistency from 60\% to 89\%.
    \item An open-source trajectory dataset \& reproducible benchmark pipeline.
\end{itemize}

\section{Method}

\subsection{Dynamic Expert Routing}
We replace dense forward passes with input-aware Mixture-of-Experts (MoE) routing. Given input $x$, the router selects a subset of $k$ experts out of $N$ total experts:
\begin{equation}
    \text{Router}(x) = \text{TopK}(\text{softmax}(W x + b), k)
\end{equation}
This reduces FLOPs by $85\%$ while preserving representational capacity.

\subsection{Hierarchical Reward Estimation}
Instead of naive voting, we cluster candidate actions via semantic embeddings, then perform pairwise comparison within clusters. The final reward $r(y)$ for action $y$ is defined as:
\begin{equation}
    r(y) = \frac{1}{|C|} \sum_{c \in \text{clusters}} \text{Score}(y, c)
\end{equation}
where $\text{Score}$ measures semantic alignment with cluster representatives.

\subsection{Byzantine Aggregation}
To prevent corrupted gradients from failing agents, we apply Krum-style secure aggregation, discarding statistical outliers before global parameter synchronization.
